## 404 - Not Found

Displayed when a requested page or resource doesn't exist.

**Route:** `/404`  
**File:** `src/features/errors/not-found-error.tsx`

### Implementation

```tsx
// src/routes/(errors)/404.tsx
import { createFileRoute } from '@tanstack/react-router'
import { NotFoundError } from '@/features/errors/not-found-error'

export const Route = createFileRoute('/(errors)/404')({\n  component: NotFoundError,
})
```

### Component

```tsx
// src/features/errors/not-found-error.tsx
import { useNavigate, useRouter } from '@tanstack/react-router'
import { Button } from '@/components/ui/button'

export function NotFoundError() {
  const navigate = useNavigate()
  const { history } = useRouter()
  
  return (
    <div className='h-svh'>
      <div className='m-auto flex h-full w-full flex-col items-center justify-center gap-2'>
        <h1 className='text-[7rem] leading-tight font-bold'>404</h1>
        <span className='font-medium'>Oops! Page Not Found!</span>
        <p className='text-center text-muted-foreground'>
          It seems like the page you're looking for <br />
          does not exist or might have been removed.
        </p>
        <div className='mt-6 flex gap-4'>
          <Button variant='outline' onClick={() => history.go(-1)}>
            Go Back
          </Button>
          <Button onClick={() => navigate({ to: '/' })}>
            Back to Home
          </Button>
        </div>
      </div>
    </div>
  )
}
```

### Features

- Large, bold 404 error code
- Friendly error message
- "Go Back" button using browser history
- "Back to Home" button for navigation
- Full viewport height centering

## 500 - Server Error

General server error page for unexpected failures.

**Route:** `/500`  
**File:** `src/features/errors/general-error.tsx`

### Component

```tsx
// src/features/errors/general-error.tsx
import { useNavigate, useRouter } from '@tanstack/react-router'
import { cn } from '@/lib/utils'
import { Button } from '@/components/ui/button'

type GeneralErrorProps = React.HTMLAttributes<HTMLDivElement> & {
  minimal?: boolean
}

export function GeneralError({
  className,
  minimal = false,
}: GeneralErrorProps) {
  const navigate = useNavigate()
  const { history } = useRouter()
  
  return (
    <div className={cn('h-svh w-full', className)}>
      <div className='m-auto flex h-full w-full flex-col items-center justify-center gap-2'>
        {!minimal && (
          <h1 className='text-[7rem] leading-tight font-bold'>500</h1>
        )}
        <span className='font-medium'>Oops! Something went wrong {`:)`}</span>
        <p className='text-center text-muted-foreground'>
          We apologize for the inconvenience. <br /> 
          Please try again later.
        </p>
        {!minimal && (
          <div className='mt-6 flex gap-4'>
            <Button variant='outline' onClick={() => history.go(-1)}>
              Go Back
            </Button>
            <Button onClick={() => navigate({ to: '/' })}>
              Back to Home
            </Button>
          </div>
        )}
      </div>
    </div>
  )
}
```

### Features

- Optional `minimal` prop to hide error code and buttons
- Customizable via `className` prop
- Friendly, apologetic messaging

<Info>
The 500 error component supports a minimal mode for embedding in error boundaries.
</Info>

## Dynamic Error Routing

Shadcn Admin also includes a dynamic error route for programmatic error handling:

**File:** `src/routes/_authenticated/errors/$error.tsx`

This allows you to navigate to errors dynamically:

```tsx
navigate({ to: '/errors/$error', params: { error: '404' } })
```

## Error Boundaries

Use error pages with React Error Boundaries:

```tsx
import { GeneralError } from '@/features/errors/general-error'

class ErrorBoundary extends React.Component {
  state = { hasError: false }

  static getDerivedStateFromError(error) {
    return { hasError: true }
  }

  render() {
    if (this.state.hasError) {
      return <GeneralError minimal />
    }
    return this.props.children
  }
}
```
