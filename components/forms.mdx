Shadcn Admin provides a comprehensive form system built on [React Hook Form](https://react-hook-form.com/) with [Zod](https://zod.dev/) schema validation for type-safe forms.

## Overview

The form system includes:

- **React Hook Form** - Performant form state management
- **Zod Integration** - Type-safe schema validation
- **Form Components** - Accessible form building blocks
- **Custom Inputs** - Specialized input components
- **Error Handling** - Automatic validation messages

## Quick Start

Here's a complete form example from `src/features/auth/forgot-password/components/forgot-password-form.tsx`:

```tsx
import { useState } from 'react'
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { z } from 'zod'
import { Button } from '@/components/ui/button'
import {
  Form,
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from '@/components/ui/form'
import { Input } from '@/components/ui/input'

// Define schema
const formSchema = z.object({
  email: z.string().email('Invalid email address'),
})

export function ForgotPasswordForm() {
  const [isLoading, setIsLoading] = useState(false)

  const form = useForm<z.infer<typeof formSchema>>({
    resolver: zodResolver(formSchema),
    defaultValues: {
      email: '',
    },
  })

  function onSubmit(data: z.infer<typeof formSchema>) {
    setIsLoading(true)
    console.log(data)
    // Handle form submission
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="grid gap-4">
        <FormField
          control={form.control}
          name="email"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Email</FormLabel>
              <FormControl>
                <Input placeholder="name@example.com" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit" disabled={isLoading}>
          Continue
        </Button>
      </form>
    </Form>
  )
}
```

## Form Components

All form components are exported from `src/components/ui/form.tsx`.

### Form

The root form provider component.

```tsx
import { Form } from '@/components/ui/form'

<Form {...form}>
  <form onSubmit={form.handleSubmit(onSubmit)}>
    {/* Form fields */}
  </form>
</Form>
```

### FormField

Connects an input to React Hook Form's controller.

<ParamField path="control" type="Control" required>
  Form control from `useForm()`
</ParamField>

<ParamField path="name" type="string" required>
  Field name (must match schema)
</ParamField>

<ParamField path="render" type="function" required>
  Render function that receives field props
</ParamField>

```tsx
<FormField
  control={form.control}
  name="email"
  render={({ field }) => (
    <FormItem>
      <FormLabel>Email</FormLabel>
      <FormControl>
        <Input {...field} />
      </FormControl>
    </FormItem>
  )}
/>
```

### FormMessage

Displays validation error messages.

From `src/components/ui/form.tsx:135-153`:

```tsx
function FormMessage({ className, ...props }: React.ComponentProps<'p'>) {
  const { error, formMessageId } = useFormField()
  const body = error ? String(error?.message ?? '') : props.children

  if (!body) {
    return null
  }

  return (
    <p
      data-slot="form-message"
      id={formMessageId}
      className={cn('text-sm text-destructive', className)}
      {...props}
    >
      {body}
    </p>
  )
}
```

## Zod Schema Validation

### Basic Schema

```tsx
import { z } from 'zod'

const schema = z.object({
  username: z.string().min(3, 'Username must be at least 3 characters'),
  email: z.string().email('Invalid email address'),
  password: z.string().min(8, 'Password must be at least 8 characters'),
})
```

### Custom Error Messages

```tsx
const schema = z.object({
  email: z.email({
    error: (iss) => (
      iss.input === '' ? 'Please enter your email' : 'Invalid email format'
    ),
  }),
})
```

### Optional Fields

```tsx
const schema = z.object({
  name: z.string(),
  bio: z.string().optional(),
  age: z.number().optional(),
})
```

### Nested Objects

```tsx
const schema = z.object({
  user: z.object({
    firstName: z.string(),
    lastName: z.string(),
  }),
  address: z.object({
    street: z.string(),
    city: z.string(),
    zip: z.string(),
  }),
})
```

### Arrays

```tsx
const schema = z.object({
  tags: z.array(z.string()),
  emails: z.array(z.string().email()).min(1, 'At least one email required'),
})
```

### Conditional Validation

```tsx
const schema = z.object({
  role: z.enum(['user', 'admin']),
  adminKey: z.string().optional(),
}).refine(
  (data) => data.role !== 'admin' || data.adminKey,
  {
    message: 'Admin key is required for admin role',
    path: ['adminKey'],
  }
)
```

import { z } from 'zod'
import { Button } from '@/components/ui/button'
import {
  Form,
  FormControl,
  FormDescription,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from '@/components/ui/form'
import { Input } from '@/components/ui/input'
import { Textarea } from '@/components/ui/textarea'

  Select,

  SelectItem,
  SelectTrigger,
  SelectValue,
} from '@/components/ui/select'

        <FormField
          control={form.control}
          name="username"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Username</FormLabel>
              <FormControl>
                <Input placeholder="johndoe" {...field} />
              </FormControl>
              <FormDescription>
                This is your public display name.
              </FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        
        <FormField
          control={form.control}
          name="email"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Email</FormLabel>
              <FormControl>
                <Input type="email" placeholder="john@example.com" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        
        <FormField
          control={form.control}
          name="bio"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Bio</FormLabel>
              <FormControl>
                <Textarea
                  placeholder="Tell us about yourself"

                />

                Brief description for your profile. Max 500 characters.
              </FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        
        <FormField
          control={form.control}

## Advanced Patterns

### Dynamic Fields

```tsx
import { useFieldArray } from 'react-hook-form'
import { Plus, X } from 'lucide-react'

const schema = z.object({
  emails: z.array(
    z.object({
      value: z.string().email(),
    })
  ).min(1),
})

function DynamicForm() {
  const form = useForm<z.infer<typeof schema>>({
    resolver: zodResolver(schema),
    defaultValues: {
      emails: [{ value: '' }],
    },
  })

  const { fields, append, remove } = useFieldArray({
    control: form.control,
    name: 'emails',
  })

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
        {fields.map((field, index) => (
          <FormField
            key={field.id}
            control={form.control}
            name={`emails.${index}.value`}
            render={({ field }) => (
              <FormItem>
                <FormLabel className={index !== 0 ? 'sr-only' : ''}>
                  Emails
                </FormLabel>
                <FormControl>
                  <div className="flex gap-2">
                    <Input {...field} />
                    <Button
                      type="button"
                      variant="outline"
                      size="icon"
                      onClick={() => remove(index)}
                      disabled={fields.length === 1}
                    >
                      <X className="h-4 w-4" />
                    </Button>
                  </div>
                </FormControl>
                <FormMessage />
              </FormItem>
            )}
          />
        ))}
        
        <Button
          type="button"
          variant="outline"
          size="sm"
          onClick={() => append({ value: '' })}
        >
          <Plus className="mr-2 h-4 w-4" />
          Add Email
        </Button>
        
        <Button type="submit">Submit</Button>
      </form>
    </Form>
  )
}
```

### Server-Side Validation

```tsx
function onSubmit(data: z.infer<typeof schema>) {
  fetch('/api/submit', {
    method: 'POST',
    body: JSON.stringify(data),
  })
    .then((res) => res.json())
    .then((result) => {
      if (result.errors) {
        // Set server errors
        Object.keys(result.errors).forEach((key) => {
          form.setError(key as any, {
            type: 'server',
            message: result.errors[key],
          })
        })
      } else {
        toast.success('Success!')
      }
    })
}
```

## Best Practices

1. **Define schemas outside components** for reusability
2. **Use type inference** with `z.infer<typeof schema>`
3. **Provide clear error messages** in your schemas
4. **Use FormDescription** for helpful hints
5. **Disable submit button** during submission
6. **Reset form** after successful submission
7. **Handle server errors** with `form.setError()`
8. **Use toast notifications** for feedback
